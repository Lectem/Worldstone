231 void DCxView::refreshFrame()
232 {
233  if (currentDCx) {
234  image->show();
235  frameHeaderInfo->show();
236  int dir = directionSpinBox->value();
237  int frameInDir = frameSpinBox->value();
238 
239  if (!animationTimer->isActive()) {
240  frameHeaderInfo->setText(currentDCx->getFrameHeaderDesc(dir, frameInDir));
241  }
242  // Display the image
243  // This costs way too much when playing the animation, but enough for now
244  ImageView<const uint8_t> frameView = currentDCx->getFrameImage(dir, frameInDir);
245  if (!frameView.isValid()) {
246  qDebug() << "Failed to get a valid frame data";
247  }
248  QImage bmp(int(frameView.width), int(frameView.height), QImage::Format::Format_Indexed8);
249  // Copy line per line because scanlines must be 32bits aligned for Qt
250  for (int y = 0; y < frameView.height; ++y)
251  {
252  memcpy(bmp.scanLine(y), &frameView(0, y), frameView.width);
253  }
254  bmp.setColorCount(Palette::colorCount);
255  if (palette.isValid()) {
256  for (int i = 0; i < Palette::colorCount; ++i)
257  {
258  auto& c = palette.colors[i];
259  bmp.setColor(i, QColor(c.r, c.g, c.b).rgb());
260  }
261  }
262  else // Display grayscale if no palette
263  {
264  for (int i = 0; i < Palette::colorCount; ++i)
265  {
266  bmp.setColor(i, QColor(i, i, i).rgb());
267  }
268  }
269  image->setPixmap(QPixmap::fromImage(bmp));
270  }
271  else
272  {
273  frameHeaderInfo->hide();
274  image->hide();
275  }
276 }
