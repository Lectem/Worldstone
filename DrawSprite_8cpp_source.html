157 bool draw(int width, int height)
158 {
159  // float time = (float)((bx::getHPCounter() - m_timeOffset) /
160  // double(bx::getHPFrequency()));
161  float at[3] = {0.0f, 0.0f, 0.0f};
162  float eye[3] = {0.0f, 0.0f, -35.0f};
163 
164  // Set view and projection matrix for view 0.
165  float view[16];
166  bx::mtxLookAtRh(view, eye, at);
167 
168  float proj[16];
169  bx::mtxProjRh(proj, 60.0f, float(width) / float(height), 0.1f, 100.0f,
170  bgfx::getCaps()->homogeneousDepth);
171  bgfx::setViewTransform(0, view, proj);
172 
173  // RightHanded
174  bx::mtxOrtho(proj, 0.f, float(width), 0.f, float(height), 1000.0f, -1000.f, 0.0f,
175  bgfx::getCaps()->homogeneousDepth);
176  // bx::mtxOrthoRh(proj, -1.f, 1.f, -1.f, 1.f, 1000.0f, -1.f, 0.0f,
177  // bgfx::getCaps()->homogeneousDepth);
178  bgfx::setViewTransform(0, nullptr, proj);
179 
180  // Set view 0 default viewport.
181  bgfx::setViewRect(0, 0, 0, uint16_t(width), uint16_t(height));
182 
183  // This dummy draw call is here to make sure that view 0 is cleared
184  // if no other draw calls are submitted to view 0.
185  bgfx::touch(0);
186 
187  // Submit 1 quad
188  float mtx[16];
189  bx::mtxIdentity(mtx);
190  bx::mtxScale(mtx, 100);
191  // Translation
192  mtx[12] = 0.f;
193  mtx[13] = 0.f;
194  mtx[14] = 0.f;
195 
196  // Set model matrix for rendering.
197  bgfx::setTransform(mtx);
198 
199  // Set vertex and index buffer.
200  bgfx::setVertexBuffer(0, m_vbh);
201  bgfx::setIndexBuffer(m_ibh);
202 
203  // Bind texture
204  bgfx::setTexture(0, s_texColor, m_textureColor);
205  bgfx::setTexture(1, s_palColor, m_paletteColor);
206 
207  // Set render states.
208  bgfx::setState(0 | BGFX_STATE_DEFAULT | BGFX_STATE_PT_TRISTRIP);
209 
210  // Submit primitive for rendering to view 0.
211  bgfx::submit(0, m_program);
212  return true;
213 }
